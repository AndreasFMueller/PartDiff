\draw[->,line width=0.7pt] (0,-3.3)--(0,3.4) coordinate[label={right:$y$}];

		({0.5*(exp(\x)+exp(-\x))},{0.5*(exp(\x)-exp(-\x))});

	\draw[line width=0.7pt] ({\t},-0.05)--({\t},0.05);
	\draw[line width=0.7pt] (-0.05,{\t})--(0.05,{\t});
	\draw[line width=0.7pt] (-0.05,{-\t})--(0.05,{-\t});
