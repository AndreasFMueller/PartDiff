\begin{scope}

	\draw[color=red,line width=1pt] (0,0)--({7*cos(45-\c)},{7*sin(45-\c)});
	\draw[color=red,line width=1pt] (0,0)--({7*cos(45+\c)},{7*sin(45+\c)});

	\draw[line width=1pt,color=darkgreen]
		plot[domain=-2.5:2.5,samples=100] ({\c*exp(\x)},{\c*exp(-\x)});
}

\end{scope}
