\begin{teilaufgaben}
\item
The function $u(x)$ is defined on the interval $\Omega = [1,2]$.

The function $u(x)$ satisfies in $\Omega$ 
\[
u''(x) + x^2 = u(x)
\]
and  $u(1) = 3, u(2) = 6$ on the boundary of $\Omega$. 

Determine approximate values for $u(4/3)$ and $u(5/3)$. 

Use finite differences with $h = 1/3$. 

\item
The function $u(x,t)$ is defined on the strip
$\Omega = [0, 1] \times [0,\infty)$.

The function $u(x,t)$ satisfies in $\Omega$
\[
u_{t}(x,t) = u_{xx}(x,t)
\]
and $u(0,t) = u(1,t) = 0$ as well as $u(x,0) = \sin(\pi \cdot x)$
on the boundary of $\Omega$.

Determine approximate values for $u(1/3,1/10)$ and $u(2/3,1/10)$.

Use the method of Crank-Nicolson with $\Delta x = 1/3$ and $\Delta t = 1/20.$
\end{teilaufgaben}

\begin{loesung}
\begin{teilaufgaben}
\item
By the method of finite differences we have
\[
9 \cdot (\tilde u(5/3) - 2 \cdot \tilde u(4/3) + \tilde u(3/3)) + (4/3)^2 = \tilde u(4/3)
\]
\[
9 \cdot (\tilde u(6/3) - 2 \cdot \tilde u(5/3) + \tilde u(4/3)) + (5/3)^2 = \tilde u(5/3)
\]

This leads to a system of two linear equations:
\begin{equation}
\begin{pmatrix*}[r]
 -19 & 9 \\
   9 & -19
\end{pmatrix*}
\cdot
\begin{pmatrix}
\tilde u(4/3) \\
\tilde u(5/3)
\end{pmatrix}
=
\begin{pmatrix*}[r]
 -259/9  \\
 -511/9
\end{pmatrix*}
\tag{2P}
\end{equation}
Its solution is
\begin{equation}
(\tilde u(4/3),  \tilde u(5/3)) = (34/9 , 43/9)
\tag{\bf 1P}
\end{equation}

\item
We use the method of Crank-Nicolson with $r = 9/20$. 

\smallskip

The Crank-Nicolson-matrices are 
\begin{equation}
F
=
1/20
\cdot
\begin{pmatrix*}[c]
 58 & -9 \\
 -9 & 58
\end{pmatrix*}
\qquad \text{and}\qquad
G = 1/20 \cdot
\begin{pmatrix*}[c]
 22 & 9 \\
  9 & 22
\end{pmatrix*}
\tag{\bf 1P}
\end{equation}
We have
\[
\begin{pmatrix*}[c]
  58 & -9 \\
  -9 & 58
\end{pmatrix*}
\cdot
\begin{pmatrix}
\tilde u(1/3,1/10)  \\
\tilde u(2/3,1/10)
  \end{pmatrix}
=
\begin{pmatrix*}[c]
 22 & 9 \\
  9 & 22
\end{pmatrix*}
\begin{pmatrix}
 \sqrt{3}/2  \\
 \sqrt{3}/2
\end{pmatrix}
\]
and therefore
\begin{equation}
(\tilde u(1/3,1/10), \tilde u(2/3,1/10))
=
(0.5479, 0.5479)
\tag{\bf 1P}
\end{equation}

We have
\[
\begin{pmatrix*}[c]
58 & -9 \\
  -9 & 58
\end{pmatrix*}
\cdot
\begin{pmatrix}
\tilde u(1/3,2/10)  \\
\tilde u(2/3,2/10) 
\end{pmatrix}
=
\begin{pmatrix*}[c]
22 & 9 \\
 9 & 22
\end{pmatrix*}
\begin{pmatrix}
 0.5479  \\
 0.5479
\end{pmatrix}
\]
and therefore
\begin{equation}
(\tilde u(1/3,2/10), \tilde u(2/3,2/10)) = (0.3466, 0.3466)
\tag{\bf 1P}
\end{equation}
\end{teilaufgaben}
\end{loesung}
