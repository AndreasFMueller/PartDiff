\[
\Omega = [0, 4] \times [0,\infty)

\[
u(x,0) = 4 \cdot x - x^2 \ \ \ \text{und} \ \ \ u_t(x,0) = 3.

(1, 3), \ \  (2, 3), \ \ (3,3).
