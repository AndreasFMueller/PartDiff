	--
	cycle;

	plot[domain=-4.2:{-3.1415/4.2},samples=100] (\x,{3.1415/\x});

	plot[domain=-4.2:{-2*3.1415/4.2},samples=100] (\x,{2*3.1415/\x});

\begin{scope}

	plot[domain=3.1415:1,samples=100] (\x,{3.1415/\x})
	--

	-- cycle;

	plot[domain=3.1415:1,samples=100] (\x,{3.1415/\x})
