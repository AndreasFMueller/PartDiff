\definecolor{darkgreen}{rgb}{0,0.6,0}

\begin{document}
\def\skala{1}
\begin{tikzpicture}[>=latex,thick,scale=\skala]

\begin{scope}
\clip (-2,-2) rectangle (2,2);
\fill[color=darkgreen!10] (-2,-2) rectangle (2,2);

\end{scope}

}
\end{scope}
