On the domain $\Omega = \{ (x,y)\mid 0<x\}$, solve the partial differential
equation
\begin{equation}
\frac{\partial u}{\partial x} + y\frac{\partial u}{\partial y} = \frac{1}{2u}
\label{30000037:pde}
\end{equation}
with boundary condition
\begin{equation}
u(0,y_0) = |y_0|
\label{30000037:boundary}
\end{equation}
for $y_0\in\mathbb{R}$.

\begin{loesung}
\begin{figure}
\centering
\includeagraphics[]{solution.pdf}
\caption{Characteristics and solution of the partial differential equation
\eqref{30000037:pde}.
\label{30000037:fig}}
\end{figure}
This is a quasilinear partial differential equation 
\[
\left.
\begin{aligned}
\dot{x} &= 1 \\
\dot{y} &= y \\
\dot{u} &= \frac{1}{2u}
\end{aligned}
\quad
\right\}
\qquad\Rightarrow\qquad
\left\{
\quad
\begin{aligned}
x(t) &= t + x_0 \\
y(t) &= y_0e^t.
\end{aligned}
\right.
\]
To satisfy the boundary condition, we choose $x_0=0$ so that characteristic
curves begin on the left boundary of $\Omega$ at $(0,y_0)$ for $t=0$.

The differential equation for $u$ is
\begin{align*}
\dot{u}&=\frac{1}{2u}
&&\Rightarrow&
u\dot{u} &= \frac12
&&\Rightarrow&
1
=
2u\dot{u}
&=
\frac{d}{dt}(u^2)
&&\Rightarrow&
u^2 &= t + u_0^2
&&\Rightarrow&
u &= \!\sqrt{t + u_0^2}.
\intertext{Alternatively, this solution can be arrived at via the calculation}
\frac{du}{dt}&=\frac{1}{2u}
&&\Rightarrow&
2u\,du &= dt
&&\Rightarrow&
\int 2u\,du &= \int dt
&&\Rightarrow&
u^2 &= t + C
&&\Rightarrow&
u &= \!\sqrt{t + u_0^2}.
\end{align*}
From the boundary condition \eqref{30000037:boundary} we can derive
the initial condition $u(0) = u_0 = |y_0|$ for the function $u(t)$.
Therefore, the parameters $y_0$ and $t$ need to be eliminated from the
system
\begin{align*}
x &= t \\
y &= y_0e^t \\
u &= \!\sqrt{t + y_0^2}
\end{align*}
of equations.
The first equation allows to replace $t$ by $x$ throughout the system
of equations.
The second equation can then be solved for
\[
y_0 = ye^{-x}.
\]
Substituting this into the last equation gives the solution
\begin{equation}
u = \!\sqrt{x+y^2e^{-2x}}.
\label{30000037:solution}
\end{equation}
The solution can be verified by differentiating:
\[
\left.
\begin{aligned}
\frac{\partial u}{\partial x}
&=
\frac{1-2y^2e^{-2x}}{2u}
\\
\frac{\partial u}{\partial y}
&=
\frac{ye^{-2x}}{u}
\end{aligned}
\right\}
\quad\Rightarrow\quad
\frac{\partial u}{\partial x}
+
y\frac{\partial u}{\partial y}
=
\frac{1}{2u},
\]
which confirms that
\eqref{30000037:solution}
is the solution to the partial differential equation
\eqref{30000037:pde}.
For the boundary condition we substitute $x=0$ giving
\[
u(0,y_0) = \!\sqrt{0+y_0^2} = |y_0|,
\]
showing that the boundary condition is satisfied as well.
A graph of the solution and the characteristics is shown in
figure~\ref{30000037:fig}.
\end{loesung}

\begin{bewertung}
Quasilinear partial differential equation of first order ({\bf Q}) 1 point,
equation for characteristic curves ({\bf C}) 1 point,
solution for $x(t)$ ({\bf X}), $y(t)$ ({\bf Y}) and $u(t)$ ({\bf U})
1 point each, elimination of $t$ and $y_0$ ({\bf E}) 1 point.
\end{bewertung}
